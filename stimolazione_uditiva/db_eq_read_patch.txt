import json

def read_eq_profile(conn, eq_profile_id: int):
    """
    Ritorna: (nome, params_dict, gain_dx_dict, gain_sx_dict)
    gain_* dict con chiavi int (freq).
    """
    cur = conn.cursor()
    try:
        cur.execute(
            "SELECT nome, params_json, gain_dx_json, gain_sx_json FROM eq_profiles WHERE id=%s",
            (int(eq_profile_id),),
        )
        row = cur.fetchone()
        if not row:
            return None
        nome, params_j, dx_j, sx_j = row

        def _load(x):
            if x is None:
                return {}
            if isinstance(x, (dict, list)):
                return x
            return json.loads(x)

        params = _load(params_j)
        dx = {int(k): float(v) for k, v in _load(dx_j).items()}
        sx = {int(k): float(v) for k, v in _load(sx_j).items()}
        return str(nome), params, dx, sx
    finally:
        try:
            cur.close()
        except Exception:
            pass
