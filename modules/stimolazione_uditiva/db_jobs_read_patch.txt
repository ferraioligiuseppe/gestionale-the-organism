import json

def read_tomatis_preset(conn, preset_id: int):
    cur = conn.cursor()
    try:
        cur.execute("SELECT name, params_json FROM tomatis_presets WHERE id=%s", (int(preset_id),))
        row = cur.fetchone()
        if not row:
            return None
        name, pj = row
        if isinstance(pj, (dict, list)):
            params = pj
        else:
            params = json.loads(pj)
        return str(name), params
    finally:
        try:
            cur.close()
        except Exception:
            pass
